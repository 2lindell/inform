packagetype _plain
packagetype _code
primitive !return val -> void
primitive !plus val val -> val
primitive !minus val val -> val
primitive !jump lab -> void
primitive !modulo val val -> val
package main _plain
	kind K_number int32
	kind K_phrase_number____nothing function K_number -> void
	package MyRoutine_B _code
		local (/main/K_number) x
		local (/main/K_number) y
		code
			inv !return
				val x
			inv !return
				inv !plus
					val x
					val y
			inv !return
				inv !minus
					val x
					val y
			inv !jump
				lab .useful
			.useful
			inv !return
				inv !modulo
					val x
					val y
	constant (K_phrase_number____nothing) MyRoutine = MyRoutine_B
